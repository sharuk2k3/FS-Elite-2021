import java.util.*;

class Smallest{
    public int smallestCommonElement(int[][] mat) {
		int n = mat.length, m = mat[0].length;
		int pos[] = new int[n];
		for (int j = 0; j < m; ++j) {
			boolean found = true;
			for (int i = 1; i < n && found; ++i) {
				pos[i] = Arrays.binarySearch(mat[i], pos[i], m, mat[0][j]);
				if (pos[i] < 0) {
					found = false;
					pos[i] = -pos[i] - 1;
					if (pos[i] >= m) {
						return -1;
					}
				}
			}
			if (found) {
				return mat[0][j];
			}
		}
		return -1;
	}
	
	public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
		int m=sc.nextInt();
		int n=sc.nextInt();
		int[][] arr = new int[m][n]; 
		for(int i=0;i<m;i++)
		for(int j=0;j<n;j++)
			arr[i][j] = sc.nextInt();
		
		System.out.println( new Smallest().smallestCommonElement(arr) ); 
	} 
}
